- Tên biến nên có ý nghĩa hơn.
-> Sửa:
//Asume total distance is 2 km
let s = 2;

// First halfway = Second halfway = Wholeway / 2
let s1 = s2 = s / 2;
let v1 = 60;
let v2 = 40;
let t1 = s1 / v1;
let t2 = s2 / v2;
let vAvg = s / (t1 + t2);

console.log("Average velocity is: " + vAvg + " (km/h)");
